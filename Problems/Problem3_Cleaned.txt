/**
 *  P1: Convert to a number to absolute number
 */

function ConvertToAbsoluteNumber(n){
  if(n<0){
    return "(" + Math.abs(n) + ")";
  } else {
	return n;
  } 
}

console.log(ConvertToAbsoluteNumber(-1));
console.log(ConvertToAbsoluteNumber(0));
console.log(ConvertToAbsoluteNumber(1));
console.log(ConvertToAbsoluteNumber(10));

calculateDiscount(baseTotal,options.discount){
	if(options.discount == Null)
	return 0;

	return baseTotal*taxPercent;
}
