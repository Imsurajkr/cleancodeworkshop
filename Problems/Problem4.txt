function authenticateUser(string username, string password)
{
    return validateAccount(username, password)  
}

function validateAccount(string UserName, string Password)
{
    	var acc = getByUserName(UserName);
	if(acc==NULL || UserName=="")
	{
	return "Please enter valid credentials"
	}
	else
	{
    	return acc.Password == Password;
	}
}

function getByUserName(string u)
{
    var account = database.ExecuteQuery("Select * from users where username='{u}'");
    return account;
}
