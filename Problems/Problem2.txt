using System;
using System.Collections.Generic;

namespace CleanCodeExample
{
    class ToBeCleanedCode
    {
        static void Main(string[] args)
        {
            List<Person> People = generatePeople();

            printGreetings(People);
            
            printAmanOfficialGreetings(People);

            readLine();
        }

        private static List<Person> generatePeople(){
            List<Person> People = new List<Person>();
            People.Add(generatePerson("Aman","Gupta"));
            People.Add(generatePerson("Aman","Agarwal"));
            People.Add(generatePerson("Aman","Verma"));
            People.Add(generatePerson("Aman","Singh"));
            return People;
        }

        private static Person generatePerson(string firstName,string lastName){
            return new Person { FirstName = firstName, LastName = lastName };
        }

        private static void printGreetings(List<Person> People){
            for (int i = People.Count - 1; i >= 0; i--)
            {
                writeLine($"Hello {People[i].FirstName} {People[i].LastName})
            }
        }

        private static void printAmanOfficialGreetings(List<Person> People){
            foreach (var person in People)
            {
                if (person.FirstName == "Aman")
                    writeLine($"Hello Mr. {person.LastName}");
            }
        }

        private static void writeLine(string message){
            Console.WriteLine(message);
        }

        private static void readLine(string message){
            Console.ReadLine();
        }
    }
}
