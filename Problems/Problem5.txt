void triangles() 
{
    cout<<"Enter height for triangle ";
    int b;
    cin>>b;
    for(int i=0; i<b; i++)
    {
        for(int j=0; j<=i; j++)
        {
            cout<<j;
        }
        cout<<"\n";
    }
}

void findGreatest() 
{
    cout<<"Enter any 3 numbers ";
    int a,b,c,max;
    cin>>a;
    cin>>b;
    cin>>c;
    max = a>b?a>c?a:c:b>c?b:c;
    cout<<"Maximum of 3 numbers is "<<max<<endl;
}

void palindrome() 
{
    cout<<"Enter any number ";
    int a,b=0,temp;
    cin>>a;
    temp = a;
    while(temp!=0)
    {
        b *= 10;
        b += temp%10;
        temp = temp/10;
    }
    if(a==b)
    {
        cout<<a<<" is a palindrome"<<endl;
    }
}

int GCD(int a, int b)
{
    if(b!=0)
    {
        return GCD(b,a%b); //Recursion
    }
    else
    {
        return a;
    }
}
