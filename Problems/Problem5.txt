void printNumbersTriangle(){
    int height;
    cout<<"Enter height for triangle ";
    cin>>height;
    for(int column=0; column<height; column++)
    {
        for(int row=0; row<=column; row++)
        {
            cout<<row;
        }
        cout<<"\n";
    }
}

void findGreatestOfThreeNumbers(){
    int number1,number2,number3,maxNumber;
    
    cout<<"Enter any 3 numbers ";
    
    cin>>number1;
    cin>>number2;
    cin>>number3;

    if(number1>number2){
        maxNumber  = number1 > number3 ? number1 : number3;
    }else{
        maxNumber = number2>number3 ? number2 : number3;
    }

    cout<< "Maximum of 3 numbers is "<< maxNumber << endl;
}

void isPalindromeNumber(){
    cout<<"Enter any number ";
    int number, reverseNumber=0, tempNumber;
    cin>>number;
    tempNumber = number;
    while(tempNumber!=0)
    {
        reverseNumber *= 10;
        reverseNumber += tempNumber%10;
        tempNumber = tempNumber/10;
    }
    if(number==reverseNumber)
    {
        cout<<number<<" is a palindrome"<<endl;
    }
}

printGCD(){
    cout<<"Enter any 2 numbers ";
    int number1,number2,gcd;
    cin>>number1;
    cin>>number2;

    cout<<"GCD of "<<number1<<" and "<<number2<<" is "<<gcd<<endl;
}

int GCD(int number1, int number2)
{
    if(number2!=0)
    {
        return GCD(number2 , number1 % number2); //Recursion
    }
    else
    {
        return number1;
    }
}
